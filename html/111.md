### Система Обновлений Мобильного Приложения

Добавьте проверку обновлений:

```dart
// lib/services/update_service.dart
import 'package:package_info_plus/package_info_plus.dart';
import 'package:dio/dio.dart';
import '../config/api_config.dart';

class UpdateService {
  static const String _currentVersion = '1.0.0';
  
  /// Проверить доступность обновления с вашего сервера
  static Future<UpdateInfo?> checkForUpdate() async {
    try {
      final response = await Dio().get('${ApiConfig.baseUrl}/api/v1/app/version');
      final serverVersion = response.data['version'];
      final forceUpdate = response.data['force_update'] ?? false;
      final updateUrl = response.data['update_url'];
      
      if (_compareVersions(serverVersion, _currentVersion) > 0) {
        return UpdateInfo(
          newVersion: serverVersion,
          forceUpdate: forceUpdate,
          updateUrl: updateUrl,
          changelog: response.data['changelog'],
        );
      }
    } catch (e) {
      print('Ошибка проверки обновления: $e');
    }
    return null;
  }
  
  static int _compareVersions(String v1, String v2) {
    final parts1 = v1.split('.').map(int.parse).toList();
    final parts2 = v2.split('.').map(int.parse).toList();
    for (int i = 0; i < 3; i++) {
      if (parts1[i] > parts2[i]) return 1;
      if (parts1[i] < parts2[i]) return -1;
    }
    return 0;
  }
}

class UpdateInfo {
  final String newVersion;
  final bool forceUpdate;
  final String? updateUrl;
  final String? changelog;
  
  UpdateInfo({
    required this.newVersion,
    required this.forceUpdate,
    this.updateUrl,
    this.changelog,
  });
}
```

### Пример Интеграции Stripe

```python
# app/routers/payments_v2.py
from fastapi import APIRouter, Depends, HTTPException
import stripe
from app.core.config import settings

router = APIRouter()
stripe.api_key = settings.STRIPE_SECRET_KEY

@router.post("/create-payment-intent")
async def create_payment_intent(
    amount: int,  # в копейках (например 50000 = 500 MDL)
    student_id: int,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """Создать намерение платежа для родителя"""
    
    # Проверка прав (родитель может платить только за своих детей)
    guardian = db.query(StudentGuardian).filter(
        StudentGuardian.student_id == student_id,
        StudentGuardian.user_id == current_user.id
    ).first()
    if not guardian and current_user.role not in ['admin', 'super_admin']:
        raise HTTPException(status_code=403, detail="Нет доступа")
    
    # Создаем транзакцию в БД
    transaction = PaymentTransaction(
        student_id=student_id,
        user_id=current_user.id,
        amount=amount / 100,
        currency='MDL',
        status='pending',
        provider='stripe'
    )
    db.add(transaction)
    db.commit()
    db.refresh(transaction)
    
    # Создаем Payment Intent в Stripe
    try:
        intent = stripe.PaymentIntent.create(
            amount=amount,
            currency='mdl',
            metadata={
                'transaction_id': transaction.id,
                'student_id': student_id,
                'user_id': current_user.id
            }
        )
        
        transaction.provider_tx_id = intent.id
        transaction.provider_payload = intent
        db.commit()
        
        return {
            'client_secret': intent.client_secret,
            'transaction_id': transaction.id
        }
    except Exception as e:
        transaction.status = 'failed'
        transaction.failed_reason = str(e)
        db.commit()
        raise HTTPException(status_code=500, detail=str(e))

@router.post("/webhook/stripe")
async def stripe_webhook(
    request: Request,
    db: Session = Depends(get_db)
):
    """Обработка вебхуков от Stripe"""
    payload = await request.body()
    sig_header = request.headers.get('stripe-signature')
    
    try:
        event = stripe.Webhook.construct_event(
            payload, sig_header, settings.STRIPE_WEBHOOK_SECRET
        )
    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))
    
    # Логируем вебхук
    webhook = PaymentWebhook(
        provider='stripe',
        event_type=event['type'],
        payload=event
    )
    db.add(webhook)
    db.commit()
    
    # Обрабатываем событие
    if event['type'] == 'payment_intent.succeeded':
        intent = event['data']['object']
        tx_id = intent['metadata'].get('transaction_id')
        
        if tx_id:
            transaction = db.query(PaymentTransaction).filter(
                PaymentTransaction.id == tx_id
            ).first()
            if transaction:
                transaction.status = 'completed'
                transaction.completed_at = func.now()
                
                # Обновляем баланс студента
                student = db.query(Student).get(transaction.student_id)
                if student:
                    student.balance += transaction.amount
                    student.is_debtor = False
                
                db.commit()
                
                # Отправляем уведомление родителю
                from app.core.background_tasks import send_payment_confirmation
                await send_payment_confirmation(transaction.id)
    
    elif event['type'] == 'payment_intent.payment_failed':
        intent = event['data']['object']
        tx_id = intent['metadata'].get('transaction_id')
        
        if tx_id:
            transaction = db.query(PaymentTransaction).get(tx_id)
            if transaction:
                transaction.status = 'failed'
                transaction.failed_reason = intent.get('last_payment_error', {}).get('message')
                db.commit()
    
    return {'status': 'success'}
```

```dart
import 'package:flutter_stripe/flutter_stripe.dart';

class _PaymentScreenState extends State<PaymentScreen> {

  Future<void> _processPayment() async {
    setState(() => _isProcessing = true);
    
    try {
      // 1. Создаем Payment Intent на сервере

        studentId: widget.studentId,
        amount: (widget.amount * 100).toInt(), // Конвертируем в копейки

        paymentSheetParameters: SetupPaymentSheetParameters(
          merchantDisplayName: 'Football Academy',

          style: ThemeMode.dark,
```

## Часть 7: Стратегия Выпуска Обновлений

### Версионирование

Используйте **Semantic Versioning** (MAJOR.MINOR.PATCH):
- **1.0.0** → **1.0.1** - исправления багов
- **1.0.0** → **1.1.0** - новые функции (совместимо)
- **1.0.0** → **2.0.0** - критичные изменения (онлайн оплата)

### Процесс Обновления

**Шаг 1: Разработка v2.0**

## Часть 9: Развертывание Одной Командой

### Созданные Файлы

| Файл | Описание |
|------|----------|
| `deploy.sh` | Главный скрипт развертывания |
| `docker-compose.prod.yml` | Production Docker конфигурация |
| `Dockerfile.prod` | Оптимизированный Dockerfile |
| `nginx/nginx.prod.conf` | Production nginx с SSL |
| `.env.production.example` | Шаблон переменных окружения |

### Как Использовать

#### 1. На Сервере (первая установка)

```bash
# Загрузите проект на сервер
scp -r "football-academy-system 2" root@your-server:/opt/football-academy

# Подключитесь к серверу
ssh root@your-server

# Перейдите в директорию
cd /opt/football-academy

# Запустите установку ОДНОЙ КОМАНДОЙ:
./deploy.sh install
```

#### 2. Доступные Команды

```bash
./deploy.sh install     # 🚀 Первая установка
./deploy.sh update      # 🔄 Обновить приложение (с бэкапом)
./deploy.sh start       # ▶️  Запустить сервисы
./deploy.sh stop        # ⏹  Остановить сервисы  
./deploy.sh restart     # 🔁 Перезапустить
./deploy.sh logs        # 📋 Посмотреть логи
./deploy.sh backup      # 💾 Создать бэкап БД
./deploy.sh restore     # 📥 Восстановить из бэкапа
./deploy.sh ssl         # 🔒 Получить SSL сертификат
./deploy.sh status      # 📊 Статус сервисов
```

#### 3. Что Происходит При `./deploy.sh install`

```
1. ✅ Проверка Docker и зависимостей
2. ✅ Создание .env.production (если нет)
3. ✅ Создание директорий (uploads, logs, backups)
4. ✅ Сборка frontend (npm build)
5. ✅ Сборка Docker образов
6. ✅ Запуск всех сервисов:
   - PostgreSQL (база данных)
   - Redis (кэширование)
   - Backend (FastAPI + Gunicorn)
   - Nginx (reverse proxy)
   - Scheduler (фоновые задачи)
   - Backup (автобэкапы)
7. ✅ Миграции базы данных
8. ✅ Инициализация данных
```

### Быстрый Старт

```bash
# 1. Настройте переменные окружения
cp .env.production.example .env.production
nano .env.production  # Измените SECRET_KEY, пароль БД, домен

# 2. Запустите одной командой
./deploy.sh install

# 3. Проверьте что работает
./deploy.sh status

# 4. Получите SSL (когда DNS настроен)
./deploy.sh ssl
```
